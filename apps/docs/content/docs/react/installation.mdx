You can install using [NPM](https://www.npmjs.com/), [Yarn](https://yarnpkg.com/en/docs/install),  [PNPM](https://pnpm.io/) or [Bun](https://bun.sh/).


<Tabs groupId="tool" items={['npm', 'yarn', 'pnpm', 'bun']} defaultValue="npm">
  <Tab value="npm">
  ```bash
  npm install @stepperize/react
  ```
  </Tab>
  <Tab value="yarn">
  ```bash
  yarn add @stepperize/react
  ```
  </Tab>
  <Tab value="pnpm">
  ```bash
  pnpm install @stepperize/react
  ```
  </Tab>
  <Tab value="bun">
  ```bash
  bun add @stepperize/react
  ```
  </Tab>
</Tabs>
