Choose your preferred package manager to install `@stepperize/vue`:

<Tabs groupId="tool" items={["npm", "yarn", "pnpm", "bun"]} defaultValue="npm">

  <Tab value="yarn">```bash yarn add @stepperize/vue ```</Tab>
  <Tab value="pnpm">```bash pnpm install @stepperize/vue ```</Tab>
  <Tab value="bun">```bash bun add @stepperize/vue ```</Tab>
</Tabs>
