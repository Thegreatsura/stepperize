Choose your preferred package manager to install `@stepperize/solid`:

<Tabs groupId="tool" items={["npm", "yarn", "pnpm", "bun"]} defaultValue="npm">
  <Tab value="npm">```bash npm install @stepperize/solid ```</Tab>
  <Tab value="yarn">```bash yarn add @stepperize/solid ```</Tab>
  <Tab value="pnpm">```bash pnpm install @stepperize/solid ```</Tab>
  <Tab value="bun">```bash bun add @stepperize/solid ```</Tab>
</Tabs>
